def diss(x,y) :
  i=10
  j=0
  for col in ["bulan","kecamatan","kelurahan","rata_rata_ketinggian_air","lama_genangan","rw_terdampak","kk_terdampak","jiwa_terdampak","korban_meninggal","korban_hilang","korban_luka_berat","korban_luka_ringan","jumlah_jiwa_pengungsi_tertinggi","jumlah_tempat_pengungsian","jumlah_nilai_kerugian"] :
    if data[col][x]==data[col][y] :
      j+=1
    return (i-j)/i
